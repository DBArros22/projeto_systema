document.addEventListener("DOMContentLoaded", () => { // Botões do menu const btnClientes = document.getElementById("btnClientes"); const btnProcessos = document.getElementById("btnProcessos"); const btnRelatorios = document.getElementById("btnRelatorios"); const logoutBtn = document.getElementById("logoutAdmin"); // Navegação via JS if (btnClientes) { btnClientes.addEventListener("click", () => { window.location.href = "clientes_admin.html"; }); } if (btnProcessos) { btnProcessos.addEventListener("click", () => { window.location.href = "processos_admin.html"; }); } if (btnRelatorios) { btnRelatorios.addEventListener("click", () => { window.location.href = "relatorios_admin.html"; }); } // Logout com confirmação if (logoutBtn) { logoutBtn.addEventListener("click", () => { const confirmar = confirm("Deseja realmente sair do sistema?"); if (confirmar) { localStorage.removeItem("loggedInUser"); window.location.href = "index.html"; } }); } });
